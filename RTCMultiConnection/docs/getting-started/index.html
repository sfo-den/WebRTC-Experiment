RTCMultiConnection Getting Started Guide

  1. <script src="https://www.webrtc-experiment.com/RTCMultiConnection-v1.5.min.js"><script>

  2. Setup Video Conference using Simplest Syntax EVER!

    var connection = new RTCMultiConnection().connect();
openNewSessionButton.onclick = function() {
    connection.open();
};

  3. Setup Video Conference using full syntax!

    var connection = new RTCMultiConnection();

// easiest way to customize what you need!
connection.session = {
    audio: true,
    video: true
};

// on getting local or remote media stream
connection.onstream = function(e) {
    document.body.appendChild(e.mediaElement);
};

// check existing sessions
connection.connect();

// open new session
openNewSessionButton.onclick = function() {
    connection.open();
};

  4. Setup Multi-user file sharing along with advance progress-bars!

    <input type=file>
    var connection = new RTCMultiConnection();

connection.session = {
    data: true
};

connection.connect();

// open new session
openNewSessionButton.onclick = function() {
    connection.open();
};

// send single file or multiple files concurrently!
document.querySelector('input[type=file]').onchange = function() {
    connection.send(this.files[0]);
};
