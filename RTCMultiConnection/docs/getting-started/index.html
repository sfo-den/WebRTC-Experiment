  2. Setup Video Conference using Shortest Syntax EVER!

  3. Setup Video Conference using full syntax!

    // www.RTCMultiConnection.org/docs/constructor
var connection = new RTCMultiConnection();

// easiest way to customize what you need!
// www.RTCMultiConnection.org/docs/session
connection.session = {
    audio: true,
    video: true
};

// on getting local or remote media stream
// www.RTCMultiConnection.org/docs/onstream
connection.onstream = function(e) {
    document.body.appendChild(e.mediaElement);
};

// check existing sessions
// on getting local or remote media stream
// www.RTCMultiConnection.org/docs/connect
connection.connect();

// open new session
openNewSessionButton.onclick = function() {
    // www.RTCMultiConnection.org/docs/open
    connection.open();
};

  4. Setup Multi-user file sharing along with advance progress-bars!

    <input type=file>
    // www.RTCMultiConnection.org/docs/constructor
var connection = new RTCMultiConnection();

// www.RTCMultiConnection.org/docs/session
connection.session = {
    data: true  // this line suggests that don't get/use audio/video; 
                // only setup data connection
};

// www.RTCMultiConnection.org/docs/connect
connection.connect();

// open new session
openNewSessionButton.onclick = function() {
    // www.RTCMultiConnection.org/docs/open
    connection.open();
};

// send single file or multiple files concurrently!
document.querySelector('input[type=file]').onchange = function() {
    connection.send(this.files[0]);
};
